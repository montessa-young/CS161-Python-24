# check if a value is correct

guess = int(input("Enter a number between 1 and 10: "))

while guess != 9:
  print("That guess is incorrect, choose another number")
  guess = int(input("Enter a number between 1 and 10: "))

if guess == 9:
  print("You guessed correctly")
